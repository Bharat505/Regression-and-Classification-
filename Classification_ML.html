#Feature Selection
def feature_selection(x,y):
    
    test = SelectKBest(score_func=f_classif, k=4)
    fit = test.fit(x, np.ravel(y,order='C'))
    # summarize scores
    set_printoptions(precision=3)
    features = fit.transform(x)
    print(fit.scores_)
    dfscores = pd.DataFrame(fit.scores_)
    dfcolumns = pd.DataFrame(x.columns)
    #concat two dataframes for better visualization 
    featureScores = pd.concat([dfcolumns,dfscores],axis=1)
    featureScores.columns = ['Specs','Score']  #naming the dataframe columns
    print(featureScores.nlargest(100,'Score'))
    return x

def logistic_reg(data_train, data_test, label_train, label_test):
    logmodel = LogisticRegression(solver='lbfgs', multi_class='auto')
    logmodel.fit(data_train, np.ravel(label_train,order='C'))
    LogReg_label_predict = logmodel.predict(data_test)
    log_Mean_Absolute_Error=metrics.mean_absolute_error(label_test, LogReg_label_predict)
    log_Mean_Squared_Error=metrics.mean_absolute_error(label_test, LogReg_label_predict)
    Root_Mean_Squared_Error=np.sqrt(metrics.mean_squared_error(label_test, LogReg_label_predict))
    log_Root_Mean_Squared_Error = metrics.accuracy_score(label_test, LogReg_label_predict)
    logreg_Accuracy = metrics.accuracy_score(label_test, LogReg_label_predict)
    return logreg_Accuracy,log_Mean_Absolute_Error,log_Mean_Squared_Error,log_Root_Mean_Squared_Error
